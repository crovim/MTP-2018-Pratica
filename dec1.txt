#include <stdio.h>
#include <stdlib.h>
#include <locale.h>

int main ()
{
	int entrada;
	setlocale(LC_ALL, "portuguese");
	printf("Digite um número inteiro: ");
	scanf("%d", &entrada);
	
	if(entrada%2==0)
	{	printf("\nEste é um número par");
		if(entrada%3==0 || entrada%7==0)
		{	printf(", e múltiplo de 3 ou 7.");		}
	}
	if(entrada%2!=0)
	{	printf("\nEste é um número ímpar");
		if(entrada%5==0)
		{	printf(", e múltiplo de 5.");		}
	}
	
	return 0;
	system("pause");
}
