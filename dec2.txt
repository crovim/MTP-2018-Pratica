#include <stdlib.h>
#include <locale.h>
#include <time.h>

int main ()
{	srand(time(0));
	int entrada, guarda = 8192, pontos, soma;
	setlocale(LC_ALL, "portuguese");
	
	printf("Digite um número que seja divisor de 8192: ");
	scanf("%d", &entrada);
	if(entrada != 0)
	{if(guarda%entrada==0)
		{	pontos  += 1;	}	}
	do
	{	entrada = rand()%1361;
	}while(entrada < 1328);
	printf("\nQual o valor de %d + 101 ? ", entrada);
	scanf("%d", &soma);
	entrada += 101;
	if(soma == entrada)
	{	pontos += 1;	}
	
	entrada = rand()%100;
	printf("\nQual o valor de %d x 3 ? ", entrada);
	scanf("%d", &soma);
	entrada *= 3;
	if(soma == entrada)
	{	pontos += 1;	}
	
	printf("\nPontuação final = %d", pontos);
	return 0;
	system("pause");
}
